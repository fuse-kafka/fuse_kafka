00076     def load_fuse_kafka_event(self, string):
00077         """ Decodes a json logstash/fuse_kafka event string, i.e.:
00078                 - does a json decoding
00079                 - decodes @message and command fields 
00080                 - adds message_size-added field 
00081         
00082         string - the JSON input logstash/fuse_kafka event
00083 
00084         Example
00085             
00086             build.FuseKafkaLog().load_fuse_kafka_event(
00087               '{"command": "bXkgY29tbWFuZGU=", "@message": "bXkgbWVzc2FnZQ=="}')
00088 
00089             => {'message_size-added': 10,
00090                 u'command': 'my commande',
00091                 u'@message': 'my message'}
00092 
00093         Returns the decoded json object
00094         """
00095         event = json.loads(string)
00096         for item in ["@message", "command"]:
00097             event[item] += "=" * ((4 - len(event[item]) % 4) % 4)
00098             event[item] = base64.b64decode(event[item] + "==")
00099         event["message_size-added"] = len(event["@message"])
00100         return event

00180 def filter_link(a):
00181     """ Filter function for link flags:
00182     takes a link flag and modifies it if necessary, i.e.
00183     in the link is -lcrypto, returns a static library path instead
00184 
00185     Examples:
00186         filter_link('-lblah')
00187         => '-lblah'
00188         filter_link('-lcrypto')
00189         => '/usr/lib/x86_64-linux-gnu/libcrypto.a'
00190 
00191     Returns the new gcc option
00192     """
00193     if a != "-lcrypto": return a
00194     result = []
00195     for pattern in ["/usr/lib*/libcrypto.a", "/usr/lib*/*/libcrypto.a"]:
00196         result += glob.glob(pattern)
00197     if len(result) > 0:
00198         return result[0]
00199     else:
00200         return a
00201 def to_links(libs):
00202     """ Convert a library list to gcc link flags Prepends -l to a given list

00208 
00209     Returns the new converted list
00210     """
